<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

<script src="https://cdn.socket.io/4.7.2/socket.io.min.js"></script>

<script>

const socket = io('http://localhost:4000'); // change to deployed backend URL


const ctx = document.getElementById('chart').getContext('2d');

const chart = new Chart(ctx, {

type: 'line',

data: {

labels: [],

datasets: [

{ label: 'BTC (USD)', data: [], tension: 0.3 },

{ label: 'ETH (USD)', data: [], tension: 0.3 }

]

},

options: { responsive: true }

});


socket.on('market:update', (payload) => {

const ts = new Date(payload.ts).toLocaleTimeString();

const btc = payload.prices.bitcoin.usd;

const eth = payload.prices.ethereum.usd;


// keep labels length reasonable

if (chart.data.labels.length > 50) {

chart.data.labels.shift();

chart.data.datasets[0].data.shift();

chart.data.datasets[1].data.shift();

}


chart.data.labels.push(ts);

chart.data.datasets[0].data.push(btc);

chart.data.datasets[1].data.push(eth);

chart.update();


document.getElementById('latest').textContent = JSON.stringify(payload.prices, null, 2);

});

</script>
